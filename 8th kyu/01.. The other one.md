// Solution //
```javascript
function makeNegative(num) {
  if (num > 0)          // if number is less than 0 then it multiplies it with -1
    return num * -1
  else                  // If number is 0 or more than 0 then it return the number as it is
    return num
}
// test
console.log(makeNegative(1004))
```
